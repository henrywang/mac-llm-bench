+"""Sliding-window rate limiter backed by Redis."""
+import time
+from src.cache import redis_client
+
+
+def _window_key(identifier: str, window_start: int) -> str:
+    return f"rl:{identifier}:{window_start}"
+
+
+def check_rate_limit(identifier: str, window_seconds: int, max_requests: int) -> bool:
+    """Return True if the request is allowed, False if rate-limited."""
+    now = int(time.time())
+    window_start = now - (now % window_seconds)
+    key = _window_key(identifier, window_start)
+
+    pipe = redis_client.pipeline()
+    pipe.incr(key)
+    pipe.expire(key, window_seconds * 2)
+    results = pipe.execute()
+
+    current_count = results[0]
+    return current_count <= max_requests
+
+
+def record_request(identifier: str, window_seconds: int = 60) -> None:
+    """Increment the counter for an already-allowed request (cache-hit path)."""
+    now = int(time.time())
+    window_start = now - (now % window_seconds)
+    key = _window_key(identifier, window_start)
+    pipe = redis_client.pipeline()
+    pipe.incr(key)
+    pipe.expire(key, window_seconds * 2)
+    pipe.execute()
+
+
+def reset_rate_limit(identifier: str, window_seconds: int = 60) -> None:
+    """Remove all rate-limit keys for an identifier (test helper)."""
+    now = int(time.time())
+    # clear current window and previous window to be safe
+    for offset in [0, window_seconds]:
+        window_start = (now - offset) - ((now - offset) % window_seconds)
+        key = _window_key(identifier, window_start)
+        redis_client.delete(key)

+"""Server-side session tracking so admins can enumerate and invalidate active sessions."""
+import time
+import secrets
+import logging
+from src.cache import redis_client
+
+logger = logging.getLogger(__name__)
+
+SESSION_TTL = 86400  # 24 hours
+SESSION_PREFIX = 'session:'
+USER_SESSIONS_PREFIX = 'user_sessions:'
+
+
+def create_session(user_id: int, metadata: dict | None = None) -> str:
+    session_id = secrets.token_urlsafe(32)
+    key = f'{SESSION_PREFIX}{session_id}'
+    data = {
+        'user_id': str(user_id),
+        'created_at': str(int(time.time())),
+        'last_seen': str(int(time.time())),
+        **(metadata or {}),
+    }
+    pipe = redis_client.pipeline()
+    pipe.hset(key, mapping=data)
+    pipe.expire(key, SESSION_TTL)
+    pipe.sadd(f'{USER_SESSIONS_PREFIX}{user_id}', session_id)
+    pipe.expire(f'{USER_SESSIONS_PREFIX}{user_id}', SESSION_TTL)
+    pipe.execute()
+    logger.debug("created session %s for user %d", session_id, user_id)
+    return session_id
+
+
+def touch_session(session_id: str) -> bool:
+    key = f'{SESSION_PREFIX}{session_id}'
+    if not redis_client.exists(key):
+        return False
+    pipe = redis_client.pipeline()
+    pipe.hset(key, 'last_seen', str(int(time.time())))
+    pipe.expire(key, SESSION_TTL)
+    pipe.execute()
+    return True
+
+
+def get_session(session_id: str) -> dict | None:
+    key = f'{SESSION_PREFIX}{session_id}'
+    data = redis_client.hgetall(key)
+    return data if data else None
+
+
+def invalidate_session(session_id: str) -> None:
+    data = get_session(session_id)
+    if data:
+        user_id = data.get('user_id')
+        pipe = redis_client.pipeline()
+        pipe.delete(f'{SESSION_PREFIX}{session_id}')
+        if user_id:
+            pipe.srem(f'{USER_SESSIONS_PREFIX}{user_id}', session_id)
+        pipe.execute()
+
+
+def invalidate_all_sessions(user_id: int) -> int:
+    session_ids = redis_client.smembers(f'{USER_SESSIONS_PREFIX}{user_id}')
+    for sid in session_ids:
+        redis_client.delete(f'{SESSION_PREFIX}{sid}')
+    redis_client.delete(f'{USER_SESSIONS_PREFIX}{user_id}')
+    return len(session_ids)

+        'ip': client_ip,
+        'user_agent': request.headers.get('User-Agent', '')[:200],

+    revoke_token(token)

+        invalidate_session(session_id)
